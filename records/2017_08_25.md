# 實作Regression by Classification模型

``` bash
/src $ python anomaly-detection-embedding.py \
    --scope phm2012 \
    --name ad-phm-embedding \
    --step-size 6 \
    --hidden-size 16 \
    --embedding-size 100 \
    --symbol-size 132 \
    --batch-size 128 \
    --layer-depth 1 \
    --dropout-rate 0.1 \
    --learning-rates \
        1   1000   0.001 \
    --sample-size 128 # must >= batch_size and will be cut to match batch_size
```
